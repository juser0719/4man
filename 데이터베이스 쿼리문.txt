-- NC_Board Table Create SQL
CREATE TABLE NC_Board
(
    `NUMBER`        INT            NULL        AUTO_INCREMENT COMMENT '번호', 
    `IMFORMATION`   VARCHAR(45)    NOT NULL    COMMENT '분류(상품정보)', 
    `TITLE`         VARCHAR(45)    NOT NULL    COMMENT '제목', 
    `DATE`          VARCHAR(45)    NOT NULL    COMMENT '작성날짜(주문일)', 
    `INQUERY_TYPE`  VARCHAR(45)    NOT NULL    COMMENT '문의종류', 
    `ORDERNUMBER`   INT            NOT NULL    COMMENT '구매번호', 
    `PRODUCT_NAME`  VARCHAR(45)    NOT NULL    COMMENT '제품명', 
    `CONTENT`       VARCHAR(45)    NOT NULL    COMMENT '내용', 
    `CONDITION`     VARCHAR(45)    NOT NULL    COMMENT '상태', 
    `NAME`          VARCHAR(45)    NOT NULL    COMMENT '이름'
);

ALTER TABLE NC_Board COMMENT '게시판';

CREATE TABLE Sign_Up
(

    `NAME`         VARCHAR(45)    NOT NULL    COMMENT '이름', 
    `PASSWORD`     VARCHAR(45)    NOT NULL    COMMENT '비밀번호', 
    `GENDER`       VARCHAR(45)    NOT NULL    COMMENT '성별', 
    `EMAIL`        VARCHAR(45)    NULL        COMMENT '이메일', 
    `PASSWORD_CK`  VARCHAR(45)    NOT NULL    COMMENT '비밀번호 확인', 
    `PHONE1`       INT            NULL        COMMENT '전화번호1', 
    `PHONE2`       INT            NULL        COMMENT '전화번호2', 
    `PHONE3`       INT            NULL        COMMENT '전화번호3', 
    `ADDRESS`      VARCHAR(45)    NULL        COMMENT '주소', 
    PRIMARY KEY (ID)
);

ALTER TABLE Sign_Up COMMENT '로그인';
